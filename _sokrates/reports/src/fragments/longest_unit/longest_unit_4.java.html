    private void cstylecomment(){
        //if it's not at the end and it's not */ then we keep advancing anc counting lines
        advance();
        while(peekNext() != '/' && !isAtEnd()){
            if(peek() == '\n'){
                line++;
            }
            advance();
            //look ahead two letter
            if(peek() == '*' && peekNext() == '/'){
                advance();
                advance();
                break;
            }
        }
    }
